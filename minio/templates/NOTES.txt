To access MinIO from outside the cluster:

{{- if .Values.ingress.enabled }}
  API:     {{ range .Values.ingress.hosts }}http://{{ .host }}{{ end }}
{{- else }}
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "minio.fullname" . }} 9000:{{ .Values.service.ports.api }}
{{- end }}
{{- if and .Values.console.enabled .Values.consoleIngress.enabled }}
  Console: {{ range .Values.consoleIngress.hosts }}http://{{ .host }}{{ end }}
{{- else if .Values.console.enabled }}
  kubectl port-forward --namespace {{ .Release.Namespace }} svc/{{ include "minio.fullname" . }} 9001:{{ .Values.service.ports.console }}
{{- end }}

To get your credentials:

  export MINIO_ROOT_USER=$(kubectl get secret {{ include "minio.secretName" . }} -n {{ .Release.Namespace }} -o jsonpath='{.data.root-user}' | base64 -d)
  export MINIO_ROOT_PASSWORD=$(kubectl get secret {{ include "minio.secretName" . }} -n {{ .Release.Namespace }} -o jsonpath='{.data.root-password}' | base64 -d)
